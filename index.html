import streamlit as st import pandas as pd # 페이지 설정 st.set_page_config(page_title="제약 자재 공정 대시보드", layout="wide") @st.cache_data def load_and_classify(url): # 데이터 로드 df = pd.read_csv(url) # 배치별로 공정 리스트 추출 (컬럼명은 실제 데이터에 맞게 '자재명', '배치번호', '공정명'으로 가정) # 실제 데이터의 컬럼명에 따라 아래 이름을 수정하세요. grouped = df.groupby(['자재명', '배치번호'])['공정명'].apply(list).reset_index() def classify(process_list): p_str = "-".join([p.strip() for p in process_list]) # 1. 인쇄정: 과립-타정-코팅-선별-인쇄 if "인쇄" in p_str: return "인쇄정" # 2. 코팅정: 과립-타정-코팅-선별 (인쇄 없음) elif "코팅" in p_str: return "코팅정" # 3. 나정: 과립-타정-선별 (코팅/인쇄 없음) elif "과립" in p_str and "타정" in p_str and "선별" in p_str: return "나정" else: return "기타" grouped['분류'] = grouped['공정명'].apply(classify) return grouped # 데이터 불러오기 URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vTxJoIbArdj4_BDnV6wxNgLd3k55kbagI1rIdqqBBW5WMb63lHFh3K-UWoPpBwbUeRXjt9B5ieNXa05/pub?gid=240982560&single=true&output=csv" try: data = load_and_classify(URL) st.title("💊 자재 공정 분류 대시보드") st.markdown("---") # 상단 요약 지표 (Metrics) col1, col2, col3 = st.columns(3) col1.metric("코팅정", len(data[data['분류'] == '코팅정'])) col2.metric("나정", len(data[data['분류'] == '나정'])) col3.metric("인쇄정", len(data[data['분류'] == '인쇄정'])) st.write("### 분류별 자재 내역 확인") st.info("아래 탭을 클릭하면 해당 분류의 상세 자재 및 배치 내역이 나타납니다.") # 탭 구성 tab1, tab2, tab3 = st.tabs(["✨ 코팅정 내역", "⚪ 나정 내역", "🖨️ 인쇄정 내역"]) with tab1: st.subheader("코팅정 (과립-타정-코팅-선별)") target_data = data[data['분류'] == '코팅정'] if not target_data.empty: st.dataframe(target_data[['자재명', '배치번호', '공정명']], use_container_width=True) else: st.write("해당하는 데이터가 없습니다.") with tab2: st.subheader("나정 (과립-타정-선별)") target_data = data[data['분류'] == '나정'] if not target_data.empty: st.dataframe(target_data[['자재명', '배치번호', '공정명']], use_container_width=True) else: st.write("해당하는 데이터가 없습니다.") with tab3: st.subheader("인쇄정 (과립-타정-코팅-선별-인쇄)") target_data = data[data['분류'] == '인쇄정'] if not target_data.empty: st.dataframe(target_data[['자재명', '배치번호', '공정명']], use_container_width=True) else: st.write("해당하는 데이터가 없습니다.") except Exception as e: st.error(f"데이터를 불러오는 중 오류가 발생했습니다: {e}") st.write("CSV URL의 접근 권한이나 컬럼명을 확인해 주세요.")
