import streamlit as st import pandas as pd # 1. 페이지 설정 (넓게 보기) st.set_page_config(page_title="제약공정 대시보드", layout="wide") # 2. 커스텀 디자인 (카드 스타일 및 배경색) st.markdown(""" """, unsafe_with_html=True) @st.cache_data def get_data(url): df = pd.read_csv(url) df.columns = [c.strip() for c in df.columns] # 배치별 분류 로직 grouped = df.groupby(['자재명', '배치번호'])['공정명'].apply(list).reset_index() def classify(procs): p_str = "|".join([str(p) for p in procs]) if "인쇄" in p_str: return "인쇄정" if "코팅" in p_str: return "코팅정" return "나정" grouped['분류'] = grouped['공정명'].apply(classify) return grouped URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vTxJoIbArdj4_BDnV6wxNgLd3k55kbagI1rIdqqBBW5WMb63lHFh3K-UWoPpBwbUeRXjt9B5ieNXa05/pub?gid=240982560&single=true&output=csv" try: data = get_data(URL) # 상단 타이틀 st.title("🏭 생산 자재별 배치 현황 대시보드") st.write("실시간 공정 분류 및 배치 내역 조회 시스템") st.markdown("
", unsafe_with_html=True) # --- 상단 섹션: 큼직한 숫자 카드 (KPI) --- counts = data['분류'].value_counts() col1, col2, col3, col4 = st.columns(4) with col1: st.metric(label="📊 총 배치 수", value=f"{len(data)}건") with col2: st.metric(label="✨ 코팅정", value=f"{counts.get('코팅정', 0)}건", delta="Coated") with col3: st.metric(label="⚪ 나정", value=f"{counts.get('나정', 0)}건", delta="Uncoated") with col4: st.metric(label="🖨️ 인쇄정", value=f"{counts.get('인쇄정', 0)}건", delta="Printed") st.markdown("
", unsafe_with_html=True) # --- 하단 섹션: 클릭하면 쫙 나오는 구조 --- st.subheader("🔍 분류별 상세 자재 내역") # 탭으로 깔끔하게 나누기 (이미지 스타일) tab_labels = ["✨ 코팅정 상세", "⚪ 나정 상세", "🖨️ 인쇄정 상세"] tabs = st.tabs(tab_labels) for i, label in enumerate(["코팅정", "나정", "인쇄정"]): with tabs[i]: filtered = data[data['분류'] == label].copy() if not filtered.empty: # 공정 순서 보기 좋게 가공 filtered['공정 흐름'] = filtered['공정명'].apply(lambda x: " ➔ ".join(x)) # 테이블 출력 (이미지처럼 깔끔하게) st.dataframe( filtered[['자재명', '배치번호', '공정 흐름']], use_container_width=True, hide_index=True ) else: st.info(f"현재 {label} 데이터가 없습니다.") except Exception as e: st.error("데이터 로드 실패! 컬럼명이나 URL을 확인해주세요.")
